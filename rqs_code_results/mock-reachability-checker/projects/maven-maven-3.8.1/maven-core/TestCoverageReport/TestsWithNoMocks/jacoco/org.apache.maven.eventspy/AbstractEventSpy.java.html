/**
 * A skeleton eventspy that does nothing other than helping implementors.
 * @since 3.0.2
 */
public abstract class AbstractEventSpy
    implements EventSpy
{

    public void init( Context context )
        throws Exception
    {
    }

    public void onEvent( Object event )
        throws Exception
    {
    }

    public void close()
        throws Exception
    {
    }

}
